// üîç SCRIPT DE DETECCI√ìN DEL TOGGLE EN
function testEnToggleDetection() {
    console.log('üîç STARTING EN TOGGLE DETECTION TEST...');
    
    // Buscar elementos EN m√°s espec√≠ficamente
    const potentialEnElements = document.querySelectorAll('span, button, div, label, *');
    let foundElements = [];
    
    for (const element of potentialEnElements) {
        const text = element.textContent?.trim().toUpperCase();
        if (text === 'EN') {
            const rect = element.getBoundingClientRect();
            if (rect.width > 0 && rect.height > 0 && rect.top >= 0 && rect.top < 200) {
                // Verificar estado activo con m√∫ltiples m√©todos
                const parent = element.parentElement;
                const grandParent = parent?.parentElement;
                
                const isActiveChecks = [
                    element.getAttribute('data-state') === 'checked',
                    element.getAttribute('aria-checked') === 'true',
                    parent?.getAttribute('data-state') === 'checked',
                    parent?.getAttribute('aria-checked') === 'true',
                    grandParent?.getAttribute('data-state') === 'checked',
                    grandParent?.getAttribute('aria-checked') === 'true',
                    element.closest('[data-state="checked"]') !== null,
                    element.closest('[aria-checked="true"]') !== null
                ];
                
                const isActive = isActiveChecks.some(check => check === true);
                
                foundElements.push({
                    element: element,
                    text: element.textContent,
                    tag: element.tagName,
                    active: isActive,
                    position: { top: rect.top, left: rect.left, width: rect.width, height: rect.height },
                    checks: isActiveChecks,
                    className: element.className,
                    id: element.id,
                    parentTag: parent?.tagName,
                    parentClass: parent?.className
                });
                
                console.log('üîç EN Element found:', {
                    text: element.textContent,
                    tag: element.tagName,
                    active: isActive,
                    position: { top: rect.top, left: rect.left },
                    checks: isActiveChecks,
                    element: element
                });
            }
        }
    }
    
    console.log('üìä SUMMARY:', {
        totalEnElementsFound: foundElements.length,
        activeElements: foundElements.filter(el => el.active),
        inactiveElements: foundElements.filter(el => !el.active)
    });
    
    const activeToggle = foundElements.find(el => el.active);
    if (activeToggle) {
        console.log('‚úÖ ACTIVE EN TOGGLE FOUND!', activeToggle);
        return 'en';
    } else {
        console.log('‚ùå NO ACTIVE EN TOGGLE FOUND');
        return 'es';
    }
}
